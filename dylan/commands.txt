# tmux
# Reattach to session:
tmux attach -t <screen number>

# Kill session
ctrl+a+b -> : then type 'kill-session'

# Create new session with name
tmux new -s <name>

# Detach session
ctrl+a+b -> : then type 'detach'

# List tmux sessions
'tmux ls'

`ctrl-a esc`
